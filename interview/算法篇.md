# 算法
1.0、二叉树遍历
```
function TreeNode(val){
    this.val = val;
    this.left = null;
    this.right = null;
}
```
1.0.1、后序（递归版）
```js

function visit(root){
    if(root){
        visit(root.left);
        visit(root.right);
        console.log(root.val);
    }
}
```
1.0.2、中序
```js
function inOrder(root){
    if(root){
        inOrder(root.left);
        console.log(root.val);
        inOrder(root.right);
    }
}
```
1.0.3、先序
```js
function perOrder(root){
    if(root){
        console.log(root.val);
        perOrder(root.left);
        perOrder(root.right);
    }
}
```
